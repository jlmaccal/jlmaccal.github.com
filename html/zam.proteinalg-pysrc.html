 67      #center 
 68      p1 = Pos1 - numpy.mean(Pos1, axis=0) 
 69      p2 = Pos2 - numpy.mean(Pos2, axis=0) 
 70      d1, dummy = numpy.shape(p1) 
 71      #calculate E0 
 72      E0 = numpy.sum(p1*p1, axis=None) + numpy.sum(p2*p2, axis=None) 
 73      #calculate correlation matrix 
 74      C = numpy.dot(numpy.transpose(p2), p1) 
 75      #get singular value decomp 
 76      V, S, Wt = numpy.linalg.svd(C) 
 77      #check determinants 
 78      Vdet = numpy.linalg.det(V) 
 79      Wtdet = numpy.linalg.det(Wt) 
 80      #if it's a reflection, reflect along lowest eigenvalue 
 81      if Vdet*Wtdet < 0.: S[-1] = -S[-1] 
 82      #compute rmsd 
 83      rmsdsq = (E0 - 2. * numpy.sum(S)) / float(d1) 
 84      rmsdsq = max(rmsdsq, 0.) 
 85      return numpy.sqrt(rmsdsq)
