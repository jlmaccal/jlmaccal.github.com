  1  #!/usr/bin/env python

  3  """ 
  4  Standalone replica exchange.

  7  import zam.utility 
  8  Logger = zam.utility.SetupLogging(__name__) 
  9   
 10  import math 
 11  import socketring 
 12  import sequence 
 13  import rexsock 
 14   
 15  #SHORTCUT SETTINGS: 
 16  #Modify these variables without having to change code below: 
 17   
 18  Seq = "ACE GLY GLY GLY NHE"  #capped sequence 
 19  MinTemp = 270.0    #minimum replica temperature  
 20  MaxTemp = 700.0    #maximum replica temperature 
 21  NCycle = 1000      #total number of md/exchange cycles 
 22  NSteps = 5000      #number of md steps per cycle 
 23  NRep = 18          #total number of replicas 
 24  CapSeq = False     #whether or not to cap the sequence with ACE and NME 
 25  Restart = True     #False means ignore any restart data if it is there; 
 26                     #True means load if found; otherwise, start anew 
 27   
 28   
 29  #MAIN PROGRAM 
 30   
 31  #program globals (don't change) 
 32  kB= (1.381 * 6.02214) / 4184.0 
 33  Temp = [] 
 34   
 35  #convert seq to a list 
 36  Seq = sequence.SeqToList(Seq) 
 37   
 38  #This defines the acceptance function for swaps, used by the rexsock module. 
 39  #a and b are the simulations of the individual "states" (e.g., temperatures). 
 40  #The example's variables TEMPSET and EPOT are the setpoint temperature and 
 41  #potential energy.  The suffixes "1" and "2" indicate those variables at the 
 42  #beginning and end of the trajectory segment.  Other variables available are: 
 43  #EKIN (kinetic energy), EEL (electrostatic energy), EREST (restraint energy), 
 44  #EVDW (van der Waals energy), ESURF (surface energy), EBOND (bond energy), 
 45  #EANG (angle energy), EDIH (dihedral energy), TIME (simulation time), TEMP 
 46  #(kinetic temperature), PRES (virial pressure), all of which have suffixes 
 47  #of either "1" or "2". 

 48 -def PaccFunc(a, b): 

 49      dBeta = 1./(kB * a["TEMPSET"]) - 1./(kB * b["TEMPSET"]) 
 50      dE = a["EPOT2"] - b["EPOT2"] 
 51      return math.exp( min([0., dE*dBeta]) ) 

 52   
 53   
 54  #This is the main loop of the program. 
 55  if __name__ == "__main__": 
 56   
 57      #This creates a socket ring with a master server and clients; the 
 58      #file containing ring descriptors is saved in path "." 
 59      sr = socketring.RingClass(".") 
 60   
 61      #See if we are the main server 
 62   
 63      if sr.IsServer:   
 64   
 65          #This initializes the replica exchange program in the path "." and 
 66          #designates the first computer to run the program as the main server 
 67          #which will dole out job tasks.  If you want to monitor the values 
 68          #of specific variables in each state (possibilities listed above), you 
 69          #can add a third argument to the RexClass constructor which is a dict 
 70          #of variable name (key) + file to save in (value) pairs, for example: 
 71          #rx = rexsock.RexClass(NRep, ".", {"TEMP":"temp.txt", "PRES":"pres.txt"}) 
 72   
 73          rx = rexsock.RexClass(SockRing = sr, BasePath = ".", N = NRep, Restart = Restart) 
 74   
 75          #tell the socket ring how many client processors we need 
 76          sr.MaxNeeded = NRep 
 77   
 78          #Setup the progress.txt file 
 79          ProgFile = "progress.txt"     
 80          file(ProgFile,"w").write("starting up...\n") 
 81   
 82          #Setup the temperatures. 
 83          if rx.NReplica == 1: 
 84              Temp = [MinTemp] 
 85          else: 
 86              dlnT = (math.log(MaxTemp)-math.log(MinTemp)) / float(rx.NReplica-1) 
 87              Temp = [MinTemp*math.exp(float(i)*dlnT) for i in range(0,rx.NReplica)] 
 88   
 89          #If a restart was not detected, build the systems up from scratch. 
 90          #The function SimIter will iterate a function over each state 
 91          #replica.  SimSetVar will set a simulation variable in each replica 
 92          #using an array whose size is the number of replicas.  SimFarm will 
 93          #iterate a function string over each replica by allocating the task 
 94          #to the available processors. 
 95          if not rx.Restart: 
 96              rx.SimIter(lambda s: s.SysInitSeq(Seq, Cap = CapSeq)) 
 97              rx.SimSetVar("STEPSMD", [NSteps for i in range(0,rx.NReplica)]) 
 98              rx.SimSetVar("TEMPSET", Temp) 
 99              rx.SimIter(lambda s: s.SysBuild()) 
100              rx.SimFarm("SIM.RunMin()") 
101   
102          #Run the cycles.  Labels is used to identify the states in 
103          #the progress file.  rx.StartRex sets up everything necessary 
104          #for the rex simulation. 
105          Labels = ["%.1f K" % (t,) for t in Temp] 
106          rx.StartRex(PaccFunc, NCycle, Labels, ProgFile) 
107   
108          #Continue running until the REX simulation is finished 
109          while not rx.RunRex(): 
110              #let the socket ring run tasks for 1 second 
111              sr.RunTasks(1.) 
112   
113          #Close down the socket ring and tell clients to go home. 
114          sr.Finish() 
115   
116      #If the current processor is not the main server, run as a client and 
117      #connect to the server. 
118      else: 
119   
120          sr.RunClient()
